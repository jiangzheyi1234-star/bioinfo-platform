# 集成分析工作台 - 代码实现分析

## 📌 功能概述

这是一个多病原体引物设计的集成分析工作台，用户可以：
1. 查看已完成的引物设计执行结果
2. 查看推荐的引物对及其详细信息
3. 从远程服务器加载引物设计结果
4. 在同一工作台中集成多个分析能力

```python
query = """
SELECT e.execution_id, e.tool_id, e.sample_id, e.parameters, 
       e.created_at, e.completed_at, s.name AS sample_name
FROM executions e
LEFT JOIN samples s ON s.sample_id = e.sample_id
WHERE e.status = 'completed' AND e.tool_id IN ('primer_design')
```

#### 步骤 D：解析数据

```python
def _parse_primer_result_text(content: str) -> list[dict]:
    rows = []
    for line in content.splitlines():
        parts = line.strip().split("\t")  # TSV 制表符分隔
        if len(parts) < 6:
            continue
        rows.append({
            "pathogen": parts[0],
            "region_id": parts[1],
            "forward_primer": parts[2],
            "reverse_primer": parts[3],
            "position": parts[4],
            "amplicon": parts[5],
        })
    return rows
```

#### 步骤 E：统计信息

```python
# 统计候选数量（读取远端文件行数）
```

### JavaScript 渲染流程 (app_galaxy.js)

```javascript
// 1. 加载集成工作台配置
function loadIntegratedWorkbench(forceRefresh = false) {
    bridge.get_integrated_workbench_config(function(config) {
        integratedWorkbench = config;
        renderIntegratedFeatureList(config.features);
        
        // 如果有 primer_design 功能，初始化显示
        const primerView = integratedWorkbench.views?.primer_design;
        if (primerView) {
            selectedIntegratedFeatureId = 'primer_design';
            renderIntegratedPrimerView(primerView);
        }
    });
}

// 2. 渲染功能列表（左侧菜单）
function renderIntegratedFeatureList(features) {
    const container = document.getElementById('integrated-feature-list');
    container.innerHTML = '';
    
    features.forEach(feature => {
        const item = document.createElement('div');
        item.className = `integrated-feature-item ${feature.status === 'active' ? 'active' : 'disabled'}`;
        item.innerHTML = `
            <h4>${feature.name}</h4>
            <p>${feature.description}</p>
            ${feature.status === 'placeholder' ? '<span class="badge">即将推出</span>' : ''}
        `;
        item.addEventListener('click', () => {
            document.querySelectorAll('.integrated-feature-item').forEach(el => el.classList.remove('active'));
            item.classList.add('active');
            selectedIntegratedFeatureId = feature.id;
            renderIntegratedPrimerView(integratedWorkbench.views[feature.id]);
        });
        container.appendChild(item);
    });
}

// 3. 渲染引物设计视图
function renderIntegratedPrimerView(view) {
    // 更新标题和状态
    document.getElementById('integrated-title').textContent = view.title;
    document.getElementById('integrated-subtitle').textContent = view.description;
    document.getElementById('integrated-status-chip').textContent = view.status.label;
    
    // 渲染参数
    renderParametersSection(view.parameters);
    
    // 渲染摘要统计
    renderSummaryCards(view.summary);
    
    // 渲染结果表格
    renderResultsTable(view.columns, view.rows);
    
    // 渲染工件列表
    renderArtifacts(view.artifacts);
    
    // 绑定远程结果目录加载
    if (view.remote_result_dir) {
        document.getElementById('remote-primer-result-dir').value = view.remote_result_dir;
    }
}

// 4. 渲染结果表格
function renderResultsTable(columns, rows) {
    const tbody = document.getElementById('integrated-results-tbody');
    tbody.innerHTML = '';
    
    rows.forEach(row => {
        const tr = document.createElement('tr');
        tr.innerHTML = columns.map(col => `
            <td>${row[col.key] || '-'}</td>
        `).join('');
        tbody.appendChild(tr);
    });
}

// 5. 渲染摘要统计
function renderSummaryCards(summary) {
    const container = document.querySelector('.summary-grid');
    container.innerHTML = summary.map(item => `
        <div class="summary-card">
            <div class="summary-value">${item.value}</div>
            <div class="summary-label">${item.label}</div>
            <div class="summary-tone" data-tone="${item.tone}"></div>
        </div>
    `).join('');
}

// 6. 从远程目录加载结果
function loadRemotePrimerResults() {
    const remoteDir = document.getElementById('remote-primer-result-dir').value.trim();
    if (!remoteDir) {
        alert('请先输入远程结果目录');
        return;
    }
    
    bridge.get_primer_view_from_result_dir(remoteDir, function(view) {
        if (view) {
            renderIntegratedPrimerView(view);
        } else {
            alert(`无法加载 ${remoteDir}，请检查路径是否正确`);
        }
    });
}
```

## 🔌 Python-JavaScript 通信

### 通过 QWebChannel 的接口方法

在 [detection_page_web.py](bio_ui/ui/pages/detection_page_web.py) 中定义（暴露给 JavaScript）：

```python
@pyqtSlot(result=dict)
def get_integrated_workbench_config(self) -> dict:
    """获取集成工作台配置（包含动态加载的实时数据）"""
    if not self.main_window:
        return self._base_integrated_workbench_config()
    
    # 尝试从最新执行加载实时数据
    live_view = self._get_live_primer_design_view()
    if live_view:
        config = self._base_integrated_workbench_config()
        config["views"]["primer_design"] = live_view
        return config
    
    return self._base_integrated_workbench_config()

@pyqtSlot(str, result=dict)
def get_primer_view_from_result_dir(self, remote_result_dir: str) -> dict | None:
    """从指定的远程目录加载引物设计结果"""
    return self._build_primer_view_from_result_dir(remote_result_dir)
```

## 🔐 关键技术点

1. **QWebChannel** - Python 和 JavaScript 通过 WebSocket 通信
2. **远程 SSH 执行** - 读取远端服务器上的文件
3. **SQLite 查询** - 从项目数据库获取执行历史
4. **文件路径追踪** - DataRegistry 维护输入输出关系
5. **延迟加载** - QWebEngineView 仅在需要时加载（避免启动时间过长）
